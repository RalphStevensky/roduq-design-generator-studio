// Initialize Praxis SDK
import { Praxis } from "@praxis/sdk";

const client = new Praxis({
  apiKey: process.env.PRAXIS_KEY,
  region: "eu-frankfurt",        // EU data residency
});

// Subscribe to portfolio risk updates
client.risk.subscribe("portfolio_id_001", (event) => {
  console.log(`VaR 95%: $${event.var95.toLocaleString()}`);
  console.log(`Beta: ${event.beta.toFixed(3)}`);
});

// Place TWAP order
const order = await client.execute({
  symbol: "AAPL",
  side: "buy",
  quantity: 10000,
  strategy: "twap",
  window: "30min",
});
